```python
def get_songs_generator(self, artist_name: str, max_songs: int) -> Generator:
    """Генератор для пошаговой загрузки песен (экономия памяти)"""
    for i, song in enumerate(artist.songs):
        yield song, i + 1

        if self.songs_since_gc >= self.gc_interval:

class ResourceMonitor:
    def check_memory_limit(self) -> bool:
        current_memory = self.get_memory_usage()
        return current_memory > self.memory_limit_mb

def _calculate_lyrics_quality(self, lyrics: str) -> float:
    """Метрика качества текста на основе длины, разнообразия слов"""
    score = 0.0
    words = lyrics.split()
    
    # Длина и разнообразие
    if len(words) > 50: score += 0.3
    unique_ratio = len(set(words)) / len(words)
    score += unique_ratio * 0.3
    
    return min(score, 1.0)
```
